import java.time.Duration;
import java.time.LocalDateTime;
import java.util.Arrays;
import java.util.Comparator;

public class QuickSort<T extends Comparable<T>> implements IOrdenador<T> {

    private long comparacoes;
    private long movimentacoes;
    private LocalDateTime inicio;
    private LocalDateTime termino;

    public QuickSort() {
        comparacoes = 0;
        movimentacoes = 0;
    }

    @Override
    public T[] ordenar(T[] dados) {
        // Usa a ordem natural (compareTo da classe Produto)
        return ordenar(dados, (a, b) -> a.compareTo(b));
    }

    @Override
    public T[] ordenar(T[] dados, Comparator<T> comparador) {
        T[] dadosOrdenados = Arrays.copyOf(dados, dados.length);
        
        
        comparacoes = 0;
        movimentacoes = 0;
        
        inicio = LocalDateTime.now();

        
        executarQuickSort(dadosOrdenados, 0, dadosOrdenados.length - 1, comparador);

        termino = LocalDateTime.now();

        return dadosOrdenados;
    }

    
    private void executarQuickSort(T[] vet, int esq, int dir, Comparator<T> c) {
        if (esq < dir) {
            
            int pivoIndice = particao(vet, esq, dir, c);
            
            // Ordena recursivamente a esquerda e a direita do pivô
            executarQuickSort(vet, esq, pivoIndice - 1, c);
            executarQuickSort(vet, pivoIndice + 1, dir, c);
        }
    }

    
    private int particao(T[] vet, int esq, int dir, Comparator<T> c) {
        T pivo = vet[esq]; // Definindo o primeiro elemento como pivô
        int i = esq + 1;
        int j = dir;

        while (i <= j) {
            
            comparacoes++;
            if (i <= dir && c.compare(vet[i], pivo) <= 0) {
                i++;
            } 
            // Recua o ponteiro 'j' enquanto o elemento for maior que o pivô
            else if (c.compare(vet[j], pivo) > 0) {
                comparacoes++;
                j--;
            } 
                        else {
                comparacoes++;
                swap(i, j, vet);
                i++;
                j--;
            }
        }
        
        
        swap(esq, j, vet);
        return j;
    }

    private void swap(int i, int j, T[] vet) {
        if (i != j) {
            movimentacoes++;
            T temp = vet[i];
            vet[i] = vet[j];
            vet[j] = temp;
        }
    }

    public long getComparacoes() {
        return comparacoes;
    }

    public long getMovimentacoes() {
        return movimentacoes;
    }

    public double getTempoOrdenacao() {
        if (inicio == null || termino == null) return 0;
        return Duration.between(inicio, termino).toMillis();
    }
}
